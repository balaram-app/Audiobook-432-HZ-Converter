    <script>
        let appState = { 
            selectedFrequency: 432, 
            charLimit: 50000, 
            currentSpeed: 1.0,
            selectedMusic: 'off',
            sessionId: null,
            downloadUrl: null
        };

        // Backend handles voice generation - no client-side voice selection needed

        window.addEventListener('load', function() {
            updateCharCounter();
        });

        document.addEventListener('DOMContentLoaded', function() {
            updateCharCounter();
        });

        function handleFileUpload(event) {
            const file = event.target.files[0];
            if (file && file.type === 'text/plain') {
                const reader = new FileReader();
                reader.onload = function(e) {
                    document.getElementById('textInput').value = e.target.result;
                    updateCharCounter();
                };
                reader.readAsText(file);
            } else {
                alert('‚ö†Ô∏è Please upload a .txt file');
            }
        }

        function updateCharCounter() {
            const text = document.getElementById('textInput').value;
            document.getElementById('charCounter').textContent = `${text.length.toLocaleString()} / 50,000 characters`;
            
            // Update usage dashboard
            const remaining = 50000 - text.length;
            document.getElementById('usageNumber').textContent = remaining.toLocaleString();
        }

        function selectFrequency(freq, btn) {
            appState.selectedFrequency = freq;
            document.querySelectorAll('.freq-option').forEach(opt => opt.classList.remove('selected'));
            btn.classList.add('selected');
        }

        function selectBackgroundMusic(music, btn) {
            appState.selectedMusic = music;
            document.querySelectorAll('.music-option').forEach(opt => opt.classList.remove('selected'));
            btn.classList.add('selected');
        }

        async function generateAudio() {
            const text = document.getElementById('textInput').value.trim();
            if (!text) { 
                alert('‚ö†Ô∏è Please enter text first!'); 
                return; 
            }
            if (text.length > 50000) { 
                alert('‚ö†Ô∏è Text exceeds 50,000 character limit!'); 
                return; 
            }
            
            // Show loading message
            const btn = event.target;
            const originalText = btn.textContent;
            btn.textContent = '‚è≥ Generating...';
            btn.disabled = true;
            
            try {
                // Backend API URL
                const BACKEND_URL = 'https://four32hz-audiobook-backend.onrender.com';
                
                // Get selected voice
                const voiceSelect = document.getElementById('voiceSelect');
                const selectedVoice = voiceSelect.value;
                
                const response = await fetch(`${BACKEND_URL}/generate`, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json',
                    },
                    body: JSON.stringify({
                        text: text,
                        frequency: appState.selectedFrequency,
                        voice: selectedVoice,
                        background_music: appState.selectedMusic || 'off',
                        speed: appState.currentSpeed
                    })
                });
                
                const data = await response.json();
                
                if (data.success) {
                    alert(`‚úÖ ${data.message}\n\nClick "Save + Download" to get your file!`);
                    // Store session ID for download
                    appState.sessionId = data.session_id;
                    appState.downloadUrl = `${BACKEND_URL}${data.download_url}`;
                } else {
                    alert('‚ùå Generation failed: ' + (data.error || 'Unknown error'));
                }
                
            } catch (error) {
                console.error('Error:', error);
                alert('‚ùå Connection error. Make sure the backend server is running!\n\n' + 
                      'Error: ' + error.message);
            } finally {
                btn.textContent = originalText;
                btn.disabled = false;
            }
        }

        function saveAndDownload() {
            if (!appState.sessionId || !appState.downloadUrl) {
                alert('‚ö†Ô∏è Please generate audio first!');
                return;
            }
            
            // Download the file
            window.location.href = appState.downloadUrl;
            alert('üì• Downloading your 432 Hz audiobook!');
        }

        function setSpeed(speed, btn) {
            appState.currentSpeed = speed;
            document.querySelectorAll('.speed-btn').forEach(b => b.classList.remove('active'));
            btn.classList.add('active');
        }

        function generateAudio() {
            const text = document.getElementById('textInput').value.trim();
            if (!text) { alert('‚ö†Ô∏è Please enter text first!'); return; }
            if (text.length > 50000) { alert('‚ö†Ô∏è Text exceeds 50,000 character limit!'); return; }
            
            if (!voicesLoaded) {
                loadVoices();
            }
            
            const voiceSelect = document.getElementById('voiceSelect');
            const voices = speechSynthesis.getVoices();
            const selectedVoice = voices[voiceSelect.value];
            
            const utterance = new SpeechSynthesisUtterance(text);
            if (selectedVoice) {
                utterance.voice = selectedVoice;
            }
            utterance.rate = appState.currentSpeed;
            speechSynthesis.speak(utterance);
            alert('üéß Generating 432 Hz Audio...');
        }

        function showPremiumModal() { document.getElementById('premiumModal').classList.add('active'); }
        function closePremiumModal() { document.getElementById('premiumModal').classList.remove('active'); }
